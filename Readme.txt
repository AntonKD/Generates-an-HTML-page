         'valign' = 'Вертикальное выравнивание',

         'bgcolor' = 'Цвет фона'),

  array( 'text' = 'Текст'
         'cells' = 'Номера занимаемых ячеек через запятую',

         'valign' = 'Вертикальное выравнивание',
         'color' = 'Цвет текста',
         'bgcolor' = 'Цвет фона'),
  ...
где номера занимаемых ячеек - набор чисел от 1 до 9-ти через запятую, определяющих ячейку в таблице
